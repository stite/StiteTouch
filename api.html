stite.touch是一个轻量级的针对现代高级浏览器的JavaScript库。极简的API,秒速上手, 玩转手势，无侵入设计,基于原生事件,支持事件代理, 性能更好。

JS初始化：

?
/* ---示例代码----*/
 
var odemo = new StiteTouch('#demo');
  
odemo.touchstart(function(){
    alert('touchstart回调');
});
 
/* ---示例代码----*/

drag

?
/* ---示例代码----*/
 
var odiv = new StiteTouch('#div1');
var lax,lay;
odiv.touchstart(function(ev){
    ev.preventDefault();
});
odiv.drag(function(ev){
    lax = lax||0;
    lay = lay||0;
    var offx = ev.x+lax;
    var offy = ev.y+lay;
    this.style.webkitTransform = "translate3d(" + offx + "px," + offy + "px,0)";
});
odiv.dragend(function(ev){
    lax+=ev.x;
    lay+=ev.y;
});
 
/* ---示例代码----*/
    

案例:

拖拽时执行方法。注：drag和drag一般情况下是配套使用。可选择ev选回参数，封装返回x、y,xy为封装后的返装后的返回值，为当前位置为0的xy，e.preventDefault()阻止默认动作
